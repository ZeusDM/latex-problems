\begin{itemize}
	\item Show that, for any $x \in (0, \tfrac{\pi}{8} )$, such a configuration exists
	\item Determine the number of angles $x \in ( 0, \tfrac{\pi}{8} )$ for which $\angle APB =4x$
\end{itemize}
